class PluginsConfig(BaseModel):

class PluginConfig(BaseModel):

    enabled: bool = True          # "Whether the plugin is active."
    execution_mode: ExecutionMode = "enforce_ignore_error"  # enforce | enforce_ignore_error | permissive | disabled
    priority: int = 100           # ge=0, le=10000. "Execution priority (lower = runs first)."

class PluginCondition(BaseModel):

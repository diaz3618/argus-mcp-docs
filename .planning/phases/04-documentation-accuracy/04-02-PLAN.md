```python
class ReadyResponse(BaseModel):
    ready: bool = False
    reason: str = ""

class BatchResponse(BaseModel):
    status: StatusResponse
    backends: BackendsResponse
    capabilities: CapabilitiesResponse
    events: EventsResponse

class ReAuthResponse(BaseModel):
    name: str
    reauth_initiated: bool = False
    error: Optional[str] = None

class RegistrySearchResponse(BaseModel):
    servers: List[RegistryServerEntry]
    registry: str = ""
    total: int = 0

class RegistryServerEntry(BaseModel):
    name: str
    description: str = ""
    transport: str = "stdio"
    url: str = ""
    command: str = ""
    args: List[str]
    version: str = ""
    categories: List[str]

class SkillsListResponse(BaseModel):
    skills: List[SkillDetail]

class SkillDetail(BaseModel):
    name: str
    version: str = ""
    description: str = ""
    status: str = "disabled"  # enabled | disabled
    tools: int = 0
    workflows: int = 0
    author: str = ""

class SkillActionResponse(BaseModel):
    name: str
    action: str  # enabled | disabled
    ok: bool = True
```

Create this file with the following Dockerfile content (this is the custom Dockerfile referenced by custom-dockerfile-demo.yaml):
```dockerfile
# Custom Dockerfile demo for argus-mcp-catalog
# Demonstrates the container.dockerfile escape hatch for servers requiring
# custom build steps that cannot be expressed via standard ContainerConfig fields.
#
# This Dockerfile installs the MCP filesystem server with additional utilities
# (ripgrep, fd-find) that are not available as system_deps in all environments.
#
# Usage: referenced via container.dockerfile: custom-dockerfile-demo.dockerfile
# in configs/filesystem-access/custom-dockerfile-demo.yaml

FROM node:22-alpine

# Install additional search utilities beyond what system_deps provides
RUN apk add --no-cache ripgrep fd

# Create nonroot user matching Argus conventions (UID 65532)
RUN addgroup -g 65532 nonroot && adduser -u 65532 -G nonroot -H -D nonroot
RUN mkdir -p /home/nonroot /tmp && chown nonroot:nonroot /home/nonroot /tmp

# Install the MCP filesystem server
RUN npm install -g @modelcontextprotocol/server-filesystem

ENV HOME=/home/nonroot
ENV TMPDIR=/tmp

USER nonroot
WORKDIR /home/nonroot

ENTRYPOINT ["npx", "-y", "@modelcontextprotocol/server-filesystem"]
```

name: "Filesystem MCP (custom Dockerfile)"
